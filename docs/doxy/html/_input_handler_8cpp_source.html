33  void CInputHandler::toggle(SDL_Event *evt, bool pressed) {
34  switch ( evt->key.keysym.sym ) {
35  // UP
36  case SDLK_KP_8:
37  case SDLK_w:
38  case SDLK_UP:
39  this->up.toggle( pressed );
40  break;
41  // DOWN
42  case SDLK_KP_2:
43  case SDLK_s:
44  case SDLK_DOWN:
45  down.toggle( pressed );
46  break;
47  // LEFT
48  case SDLK_KP_4:
49  case SDLK_a:
50  case SDLK_LEFT:
51  left.toggle( pressed );
52  break;
53  // RIGHT
54  case SDLK_KP_6:
55  case SDLK_d:
56  case SDLK_RIGHT:
57  right.toggle( pressed );
58  break;
59  // MENU
60  case SDLK_TAB:
61  case SDLK_LALT:
62  case SDLK_RALT:
63  case SDLK_RETURN:
64  case SDLK_RETURN2:
65  case SDLK_x:
66  menu.toggle( pressed );
67  break;
68  // ATTACK
69  case SDLK_SPACE:
70  case SDLK_RCTRL:
71  case SDLK_LCTRL:
72  case SDLK_KP_0:
73  case SDLK_INSERT:
74  case SDLK_c:
75  attack.toggle( pressed );
76  break;
77  // NOTHINGS
78  default:
79  break;
80  }
81  }
